504 public LinkedNode<T>? Erase () {
505 LinkedNode<T>? prev = Prev;
506 LinkedNode<T>? next = Next;
507
508 // Fall 1: Node ist isoliert
509 if ( prev == null && next == null ) {
510 // nichts zu verbinden
511 }
512 // Fall 2: Node ist am Anfang
513 else if ( prev == null ) {
514 next!.Prev = null;
515 }
516 // Fall 3: Node ist am Ende
517 else if ( next == null ) {
518 prev!.Next = null;
519 }
520 // Fall 4: Node ist mittendrin
521 else {
522 prev!.Next = next;
523 next!.Prev = prev;
524 }
525
526#if TRACE
527 // optional: isolieren
528 Prev = null;
529 Next = null;
530#endif
531
532 return this;
533 }

654 public void SwapWith ( LinkedNode<T> node ) {
655 if ( node == null || node == this )
656 return;
657
658 // Save neighbors
659 var aPrev = this.Prev;
660 var aNext = this.Next;
661 var bPrev = node.Prev;
662 var bNext = node.Next;
663
664 // Swap Prev/Next pointers
665 this.Prev = bPrev;
666 this.Next = bNext;
667
668 node.Prev = aPrev;
669 node.Next = aNext;
670
671 // Fix neighbors
672 if ( this.Prev != null )
673 this.Prev.Next = this;
674 if ( this.Next != null )
675 this.Next.Prev = this;
676
677 if ( node.Prev != null )
678 node.Prev.Next = node;
679 if ( node.Next != null )
680 node.Next.Prev = node;

700 public void Swap ( LinkedNode<T> a, LinkedNode<T> b ) {
701 if ( a == b )
702 return;
703
704 // 1. A entfernen
705 var aPrev = a.Prev;
706 var aNext = a.Next;
707 a.Erase();
708
709 // 2. B entfernen
710 var bPrev = b.Prev;
711 var bNext = b.Next;
712 b.Erase();
713
714 // 3. A an B-Position einfügen
715 if ( bPrev != null )
716 bPrev.InsertAfter(a);
717 else if ( bNext != null )
718 bNext.InsertBefore(a);
719
720 // 4. B an A-Position einfügen
721 if ( aPrev != null )
722 aPrev.InsertAfter(b);
723 else if ( aNext != null )
724 aNext.InsertBefore(b);
725 }

832 public Pair<LinkedNode<T>, LinkedNode<T>> Splice ( LinkedNode<T> first, LinkedNode<T> last ) {
833 LinkedNode<T>? before = Prev;
834 LinkedNode<T>? after = Next;
835
836 // Entferne 'this' aus seiner Position
837 if ( before != null )
838 before.Next = after;
839 if ( after != null )
840 after.Prev = before;
841
842 // Füge [first..last] an die Stelle ein, wo 'this' war
843 if ( before != null )
844 before.Next = first;
845 first.Prev = before;
846
847 if ( after != null )
848 after.Prev = last;
849 last.Next = after;
850
851 // 'this' isolieren
852 Prev = null;
853 Next = null;
854
855 return new Pair<LinkedNode<T>, LinkedNode<T>>(this, first);

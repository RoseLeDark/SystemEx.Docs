191 public ulong Seek(SeekOrigin org, int pos) {
192 if ( m_rawBuffer.Size == 0 ) return 0;
193
194 switch ( org ) {
195
196 case SeekOrigin.Begin:
197 m_position = (ulong)System.Math.Max(pos, 0);
198 break;
199 case SeekOrigin.Current:
200 if ( pos >= 0 )
201 m_position += (ulong)pos;
202 else {
203 ulong neg = (ulong)(-pos);
204 m_position = (m_position > neg) ? (m_position - neg) : 0;
205 }
206 break;
207 case SeekOrigin.End:
208 if ( pos >= 0 ) {
209 ulong p = (ulong)pos;
210 m_position = (p > LongLength) ? 0 : (LongLength - p);
211 } else {
212 // End + negative pos = End - (-pos) = End + pos
213 ulong add = (ulong)(-pos);
214 m_position = LongLength + add; // wird unten gekappt
215 }
216 break;
217 }
218
219 if ( m_position > LongLength )
220 m_position = LongLength;
221
222 return m_position;
223 }

234 public virtual ulong WriteRange(ulong start, ulong iend, byte[] data) {
235 if ( m_isLocked ) throw new InvalidOperationException("is Locked");
236
237 // Start ungültig?
238 if ( start < 0 || start >= (ulong)m_rawBuffer.Size )
239 return 0;
240
241 // End über Größe → kappen
242 if ( iend > (ulong)m_rawBuffer.Size )
243 iend = (ulong)m_rawBuffer.Size;
244
245 // Bereich ungültig?
246 if ( iend <= start )
247 return 0;
248
249 ulong rangeLen = iend - start;
250 ulong writable = System.Math.Min((uint)rangeLen, (uint)data.Length);
251
252 for ( ulong i = 0; i < (ulong)writable; i++ )
253 m_rawBuffer[(int)(start + i)] = data[i];
254
255 if ( (start + writable) > m_maxUsedAdress ) m_maxUsedAdress = start + writable;
256
257 return (ulong)writable;
258 }

556 public int Read(byte[] buffer, int offset, int count) {
557 if ( m_isLocked ) throw new InvalidOperationException("is Locked");
558 if ( offset < 0 || count < 0 ) throw new ArgumentOutOfRangeException("offset,count");
559 ArgumentNullException.ThrowIfNull(buffer);
560 ArgumentOutOfRangeException.ThrowIfGreaterThan(offset, buffer.Length);
561
562 if ( offset + count > buffer.Length ) {
563 // Wenn der Caller mehr verlangt als im Zielbuffer Platz ist,
564 // lesen wir nur so viel, wie in den Zielbuffer passt.
565 count = buffer.Length - offset;
566 }
567
568 // Wenn die Leseposition bereits am oder jenseits des Endes ist: EOF
569 if ( (ulong)m_rawBuffer.Size <= m_position ) return 0;
570
571 // Verfügbare Bytes im Rohpuffer ab aktueller Position
572 ulong available = (ulong)m_rawBuffer.Size - m_position;
573 int toRead = (int)System.Math.Min((ulong)count, available);
574
575 if ( toRead <= 0 ) return 0;
576
577 // ReadRange erwartet position als ulong und count als uint
578 byte[]? chunk = ReadRange(m_position, (uint)toRead);
579 if ( chunk == null || chunk.Length == 0 ) return 0;
580
581 Array.Copy(chunk, 0, buffer, offset, chunk.Length);
582
583 // interne Position vorziehen
584 m_position += (ulong)chunk.Length;
585
586 return chunk.Length;
587 }

598 public int Write(byte[] buffer, int offset, int count) {
599 if ( m_isLocked ) throw new InvalidOperationException("is Locked");
600 if ( offset < 0 || count < 0 ) throw new ArgumentOutOfRangeException("offset,count");
601
602 ArgumentNullException.ThrowIfNull(buffer);
603 ArgumentOutOfRangeException.ThrowIfGreaterThan(offset, buffer.Length);
604
605
606
607 // Adjust count if caller asked for more than fits in the target buffer
608 if ( offset + count > buffer.Length ) {
609 count = buffer.Length - offset;
610 }
611
612 // If write position is already at or beyond end, nothing to write
613 if ( (ulong)m_rawBuffer.Size <= m_position ) return 0;
614
615 // How many bytes are available in the raw buffer from current position
616 ulong available = (ulong)m_rawBuffer.Size - m_position;
617 int toWrite = (int)System.Math.Min((ulong)count, available);
618
619 if ( toWrite <= 0 ) return 0;
620
621 for ( int i = 0; i < toWrite; i++ ) {
622 m_rawBuffer[(int)(m_position + (ulong)i)] = buffer[offset + i];
623 }
624
625 // advance internal position
626 m_position += (ulong)toWrite;
627
628 if ( m_position > m_maxUsedAdress ) m_maxUsedAdress = m_position;
629
630 return toWrite;
631 }

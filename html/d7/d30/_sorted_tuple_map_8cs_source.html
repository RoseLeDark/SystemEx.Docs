127 public void Sort() {
128 for ( int i = 0; i < base.Count - 1; i++ ) {
129 for ( int j = i + 1; j < base.Count; j++ ) {
130
131 CompareResult cmp = m_comparer != null
132 ? m_comparer.Compare(m_elements[i], m_elements[j])
133 : m_sort!(m_elements[i], m_elements[j]);
134
135 if ( cmp == CompareResult.AIsLargerB ) {
136 Swap(i, j);
137 }
138 }
139 }
140 }

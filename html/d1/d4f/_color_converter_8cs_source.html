61 public static ColorR8G8B8 ToColorR8G8B8(this ColorXYZ c) {
62 float X = c.X;
63 float Y = c.Y;
64 float Z = c.Z;
65
66 // XYZ → linear RGB
67 float r_lin = 3.2404542f * X - 1.5371385f * Y - 0.4985314f * Z;
68 float g_lin = -0.9692660f * X + 1.8760108f * Y + 0.0415560f * Z;
69 float b_lin = 0.0556434f * X - 0.2040259f * Y + 1.0572252f * Z;
70
71 // linear RGB → sRGB (Gamma)
72 float r = (r_lin <= 0.0031308f) ? 12.92f * r_lin : 1.055f * MathF.Pow(r_lin, 1f / 2.4f) - 0.055f;
73 float g = (g_lin <= 0.0031308f) ? 12.92f * g_lin : 1.055f * MathF.Pow(g_lin, 1f / 2.4f) - 0.055f;
74 float b = (b_lin <= 0.0031308f) ? 12.92f * b_lin : 1.055f * MathF.Pow(b_lin, 1f / 2.4f) - 0.055f;
75
76 // Clamp (sRGB muss 0..1 sein)
77 return new ColorR8G8B8(r < 0f ? 0f : (r > 1f ? 1f : r),
78 g < 0f ? 0f : (g > 1f ? 1f : g),
79 b < 0f ? 0f : (b > 1f ? 1f : b)
80 );
81 }

137 public static ColorHSV ToColorHSV(this ColorR8G8B8 x) {
138 float max = System.Math.Max(x.Red, System.Math.Max(x.Green, x.Blue));
139 float min = System.Math.Min(x.Red, System.Math.Min(x.Green, x.Blue));
140 float delta = max - min;
141
142 float h = 0f;
143 float s = 0f;
144 float v = max;
145
146 if ( delta > 0f ) {
147 s = delta / max;
148
149 if ( max == x.Red )
150 h = 60f * (((x.Green - x.Blue) / delta) % 6f);
151 else if ( max == x.Green )
152 h = 60f * (((x.Blue - x.Red) / delta) + 2f);
153 else
154 h = 60f * (((x.Red - x.Green) / delta) + 4f);
155
156 if ( h < 0f )
157 h += 360f;
158 }
159
160 return new ColorHSV(h, s, v);
161 }

165 public static ColorHWB ToColorHWB(this ColorR8G8B8 color) {
166 var c = color.AsNormalized();
167
168 float max = System.Math.Max(c.Red, System.Math.Max(c.Green, c.Blue));
169 float min = System.Math.Min(c.Red, System.Math.Min(c.Green, c.Blue));
170 float delta = max - min;
171
172 float w = min;
173 float b = 1 - max;
174 float h;
175
176 if ( delta == 0 )
177 h = 0;
178 else if ( max == c.Red )
179 h = (c.Green - c.Blue) / delta;
180 else if ( max == c.Green )
181 h = 2 + (c.Blue - c.Red) / delta;
182 else
183 h = 4 + (c.Red - c.Green) / delta;
184
185 h *= 60;
186 if ( h < 0 ) h += 360;
187
188 return new ColorHWB(h, w *= 100, b *= 100);
189 }

193 public static ColorR8G8B8 ToColorR8G8B8(this ColorHWB color) {
194 float h = Math.FMod(color.Hue, 360.0f) / 60.0f;
195 float w = color.Whiteness / 100.0f;
196 float bl = color.Blackness / 100.0f;
197
198 float v = 1 - bl;
199 float s = v == 0 ? 0 : 1 - w / v;
200 float c = v * s;
201 float x = c * (1 - System.Math.Abs( Math.FMod (h, 2.0f) - 1f));
202 float m = v - c;
203
204 float r, g, b;
205 if ( h < 1 ) { r = c; g = x; b = 0; }
206 else if ( h < 2 ) { r = x; g = c; b = 0; }
207 else if ( h < 3 ) { r = 0; g = c; b = x; }
208 else if ( h < 4 ) { r = 0; g = x; b = c; }
209 else if ( h < 5 ) { r = x; g = 0; b = c; }
210 else { r = c; g = 0; b = x; }
211
212 return new ColorR8G8B8(
213 ((r + m)),
214 ((g + m)),
215 ((b + m))
216 );
217 }

222 public static ColorR8G8B8 ToColorR8G8B8(this ColorHSV color) {
223 float C = color.V * color.S; // Chroma
224 float Hp = Math.FMod(color.H / 60f, 6f); // Hue' (0..6)
225 float X = C * (1f - System.Math.Abs(Math.FMod(Hp, 2f) - 1f));
226 float m = color.V - C;
227
228 float r, g, b;
229
230 if ( Hp < 1f ) { r = C; g = X; b = 0; }
231 else if ( Hp < 2f ) { r = X; g = C; b = 0; }
232 else if ( Hp < 3f ) { r = 0; g = C; b = X; }
233 else if ( Hp < 4f ) { r = 0; g = X; b = C; }
234 else if ( Hp < 5f ) { r = X; g = 0; b = C; }
235 else if ( Hp < 6f ) { r = C; g = 0; b = X; }
236 else { r = 0; g = 0; b = 0; }
237
238 return new ColorR8G8B8(r + m, g + m, b + m);
239 }

250 public static ColorHSL ToColorHSL(this ColorHSV color) {
251 // Lightness
252 float L = color.V * (1f - color.S * 0.5f);
253
254 // Saturation in HSL
255 float S_hsl = 0f;
256
257 if ( L > 0f && L < 1f ) {
258 float minL = (L < 0.5f) ? L : (1f - L);
259 if ( minL > 0f )
260 S_hsl = (color.V - L) / minL;
261 }
262
263 return new ColorHSL(color.H, S_hsl, L);

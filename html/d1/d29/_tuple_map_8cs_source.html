183 public List<ITuple> Findex(Compare func, object Key, object Value) {
184 List<ITuple> result = new List<ITuple>();

196 public void Traverse(TraversMode mode, int startIndex, int endIndex, Action<ITuple> func) {
197 int start = System.Math.Max(startIndex, 0);
198 int end = System.Math.Min(endIndex, m_elements.Count);
199
200 if ( mode == TraversMode.Forwards ) {
201 for ( int i = start; i < end; i++ )
202 func(m_elements[i]);
203 } else {
204 for ( int i = end; i >= start; i-- )
205 func(m_elements[i]);
206 }
207 }

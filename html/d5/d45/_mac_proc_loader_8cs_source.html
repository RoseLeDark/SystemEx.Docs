99 public static string GetLibaryPath ( string dynmodule ) {
100 string _ret = NO_PATH;
101
102 // Cheak usr/lib
103 _ret = GetFromPath("/usr/lib/", dynmodule);
104 if(_ret == NO_PATH )
105 _ret = GetFromPath("/usr/local/lib", dynmodule);
106
107 if ( _ret == NO_PATH ) {
108
109 string? env = Environment.GetEnvironmentVariable("DYLD_LIBRARY_PATH");
110 if ( env != null ) {
111 var paths = env.Split(':');
112 string _t = "";
113 foreach ( string path in paths ) {
114 _t = GetFromPath(path, dynmodule);
115 if ( _t != NO_PATH ) { _ret = _t; break; }
116 }
117 }
118 }
119
120 return _ret;
121 }
122
126 static string GetFromPath ( string path, string module ) {
127 string _ret = NO_PATH;
128 foreach ( var file in Directory.GetFiles(path) ) {
129 if ( Path.GetFileName(file) == module ) {
130 _ret = Path.Combine(path, file);
131 }
132 }
133 return _ret;
134 }

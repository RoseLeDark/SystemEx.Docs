423 public Node<T>? Find ( T value ) {
424 if ( this.Value!.Equals(value)) return this;
425 Node<T> temp = this;
426 Node<T>? ret = null;
427
428 while ( temp!.HasNext ) {
429 T? valu = temp.Value;
430 if ( valu == null ) continue;
431
432 if(valu.Equals(value)) {
433 ret = temp;
434 break;
435 }
436 temp = temp.Next!;
437 }
438
439 if( ret == null) {
440 temp = this;
441
442 while ( temp!.HasPrev ) {
443 T? valu = temp.Value;
444 if ( valu == null ) continue;
445
446 if ( valu.Equals(value) ) {
447 ret = temp;
448 break;
449 }
450 temp = temp.Prev!;
451 }
452 }
453 return ret;

480 public void Remove () {
481 Node<T>? prev = m_pNodex[PREV];
482 Node<T>? next = m_pNodex[NEXT];
483
484 // Fall 1: Node ist isoliert
485 if ( prev == null && next == null ) {
486 // nichts zu verbinden
487 }
488 // Fall 2: Node ist am Anfang
489 else if ( prev == null ) {
490 next!.m_pNodex[PREV] = null;
491 }
492 // Fall 3: Node ist am Ende
493 else if ( next == null ) {
494 prev!.m_pNodex[NEXT] = null;
495 }
496 // Fall 4: Node ist mittendrin
497 else {
498 prev!.m_pNodex[NEXT] = next;
499 next!.m_pNodex[PREV] = prev;
500 }
501
502#if TRACE
503 // optional: isolieren
504 m_pNodex[PREV] = null;
505 m_pNodex[NEXT] = null;
506#endif

514 public void Splice ( ref Node<T> first, ref Node<T> last ) {
515 // 1. Hole die alten Nachbarn
516 Node<T>? before = this.Prev;
517 Node<T>? after = this.Next;
518
519 // 2. Entferne 'this' aus seiner Position
520 if ( before != null )
521 before.Next = after;
522 if ( after != null )
523 after.Prev = before;
524
525 // 3. Füge [first..last] an die Stelle ein, wo 'this' war
526
527 // Verbinde vor dem Block
528 if ( before != null )
529 before.Next = first;
530 first.Prev = before;
531
532 // Verbinde nach dem Block
533 if ( after != null )
534 after.Prev = last;
535 last.Next = after;
536
537 // 4. 'this' ist jetzt isoliert
538 this.Prev = null;
539 this.Next = null;
540 }

547 public Node<T> Reverse () {
548 Node<T>? current = this;
549 Node<T>? newHead = null;
550
551 while ( current != null ) {
552 // Swap Prev and Next
553 Node<T>? temp = current.m_pNodex[NEXT];
554 current.m_pNodex[NEXT] = current.m_pNodex[PREV];
555 current.m_pNodex[PREV] = temp;
556
557 newHead = current;
558
559 current = current.m_pNodex[PREV];
560 }
561
562 return newHead!;
563 }

596 public void Swap ( ref Node<T> other ) {
597 if ( this == other )
598 return;
599
600 // Backup A
601 Node<T>? aPrev = this.Prev;
602 Node<T>? aNext = this.Next;
603 T? aValue = this.Value;
604
605 // Backup B
606 Node<T>? bPrev = other.Prev;
607 Node<T>? bNext = other.Next;
608 T? bValue = other.Value;
609
610 // Copy B into A
611 this.Prev = bPrev;
612 this.Next = bNext;
613 this.Value = bValue;
614
615 // Copy A into B
616 other.Prev = aPrev;
617 other.Next = aNext;
618 other.Value = aValue;
619
620 // Fix neighbors of A (null‑sicher)
621 if ( this.Prev != null )
622 this.Prev.Next = this;
623
624 if ( this.Next != null )
625 this.Next.Prev = this;
626
627 // Fix neighbors of B (null‑sicher)
628 if ( other.Prev != null )
629 other.Prev.Next = other;
630
631 if ( other.Next != null )
632 other.Next.Prev = other;
633 }

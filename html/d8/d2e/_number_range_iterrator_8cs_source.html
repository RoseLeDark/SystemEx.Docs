18using System;
19using System.Collections.Generic;

24namespace SystemEx {
30 public struct NumberRangeIterator<T> : IForwardIterator<T>
31 where T : INumber<T> {
33 private T m_current;
34
36 private readonly T m_end;
37
44 public NumberRangeIterator ( T start, T end ) {
45 m_current = start;
46 m_end = end;
47 }
48
52 public T Current => m_current;
53
57 public bool IsEnd => m_current > m_end;
58
62 public void Forward () {
63 if ( !IsEnd )
64 m_current += T.One;
65 }
66
70 public void Forward (int i) {
71 var n = i;
72 while ( n > 0 ) {
73 --n;
74 Forward();
75 }
76 }
77
81 public IIterator<T> Clone ()
82 => new NumberRangeIterator<T>(m_current, m_end);
83
88 public override bool Equals ( object? obj )
89 => obj is NumberRangeIterator<T> it &&
90 it.m_current.Equals(m_current) &&
91 it.m_end.Equals(m_end);
92
96 public override int GetHashCode ()
97 => HashCode.Combine(m_current, m_end);
98
107 public static bool operator == ( NumberRangeIterator<T> left, NumberRangeIterator<T> right ) {
108 return left.Equals(right);
109 }
110
117 public static bool operator != ( NumberRangeIterator<T> left, NumberRangeIterator<T> right ) {
118 return !left.Equals(right);
119 }
120 }

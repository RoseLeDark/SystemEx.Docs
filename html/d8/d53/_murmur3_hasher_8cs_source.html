40 public Hash32 Compute ( Array<byte> input, uint seed, Endian endian ) {
41 uint h1 = seed;
42
43 // 4‑Byte Blöcke
44 for ( ArrayRandomAccessIterator<byte> it = input.First ; it != input.End ; it.Forward(4) ) {
45 // Wenn weniger als 4 Bytes übrig sind → Tail
46 if ( Iterator.Distance(it.Clone(), input.End.Clone()) < 4 )
47 break;
48
49 byte b0 = it.Current;
50 byte b1 = Iterator.Advance<byte>(it, 1).Current;
51 byte b2 = Iterator.Advance<byte>(it, 2).Current;
52 byte b3 = Iterator.Advance<byte>(it, 3).Current;
53
54 uint k1;
55
56 if ( endian == Endian.LittleEndian ) {
57 k1 = (uint)(b0
58 | (b1 << 8)
59 | (b2 << 16)
60 | (b3 << 24));
61 } else {
62 k1 = (uint)(b3
63 | (b2 << 8)
64 | (b1 << 16)
65 | (b0 << 24));
66 }
67
68 // Murmur3‑Mixing
69 k1 *= 3432918353U;
70 k1 = RotateLeft(k1, 15);
71 k1 *= 461845907U;
72
73 h1 ^= k1;
74 h1 = RotateLeft(h1, 13);
75 h1 = h1 * 5 + 0xe6546b64;
76 }
77
78 // --- TAIL ---
79 int remainder = Iterator.Distance(input.First.Clone(), input.End.Clone()) % 4;
80
81 if ( remainder > 0 ) {
82
83 ArrayRandomAccessIterator<byte> tailIt = input.At(input.Count - remainder);
84
85 uint k1 = 0;
86
87 switch ( remainder ) {
88 case 3:
89 k1 |= (uint)tailIt.Current;
90 k1 |= (uint)Iterator.Advance<byte>(tailIt, 1).Current << 8;
91 k1 |= (uint)Iterator.Advance<byte>(tailIt, 2).Current << 16;
92 break;
93
94 case 2:
95 k1 |= (uint)tailIt.Current;
96 k1 |= (uint)Iterator.Advance<byte>(tailIt, 1).Current << 8;
97 break;
98
99 case 1:
100 k1 |= tailIt.Current;
101 break;
102 }
103
104 k1 *= 3432918353U;
105 k1 = RotateLeft(k1, 15);
106 k1 *= 461845907U;
107
108 h1 ^= k1;
109 }
110
111 h1 ^= (uint)input.Count;
112 h1 = FMix(h1);
113
114 return new Hash32(h1);
115 }

126 public Hash64 ComputeLong ( Array<byte> input, ulong seed, Endian endian ) {
127 int length = input.Count;
128 ulong h = seed;
129
130 // 8‑Byte Blöcke
131 for ( ArrayRandomAccessIterator<byte> it = input.First ; it != input.End ; it.Forward(8) ) {
132
133 if ( Iterator.Distance(it, input.End) < 8 )
134 break;
135
136 ulong k;
137
138 byte b0 = it.Current;
139 byte b1 = Iterator.Advance<byte>(it, 1).Current;
140 byte b2 = Iterator.Advance<byte>(it, 2).Current;
141 byte b3 = Iterator.Advance<byte>(it, 3).Current;
142 byte b4 = Iterator.Advance<byte>(it, 4).Current;
143 byte b5 = Iterator.Advance<byte>(it, 5).Current;
144 byte b6 = Iterator.Advance<byte>(it, 6).Current;
145 byte b7 = Iterator.Advance<byte>(it, 7).Current;
146
147 if ( endian == Endian.LittleEndian ) {
148 k = (ulong)b0
149 | ((ulong)b1 << 8)
150 | ((ulong)b2 << 16)
151 | ((ulong)b3 << 24)
152 | ((ulong)b4 << 32)
153 | ((ulong)b5 << 40)
154 | ((ulong)b6 << 48)
155 | ((ulong)b7 << 56);
156 } else {
157 k = (ulong)b7
158 | ((ulong)b6 << 8)
159 | ((ulong)b5 << 16)
160 | ((ulong)b4 << 24)
161 | ((ulong)b3 << 32)
162 | ((ulong)b2 << 40)
163 | ((ulong)b1 << 48)
164 | ((ulong)b0 << 56);
165 }
166
167 // Murmur‑like mixing
168 k *= 0x87c37b91114253d5UL;
169 k = RotateLeft64(k, 31);
170 k *= 0x4cf5ad432745937fUL;
171
172 h ^= k;
173 h = RotateLeft64(h, 27);
174 h = h * 5 + 0x52dce729;
175 }
176
177 int remainder = Iterator.Distance(input.End, input.First) % 8;
178
179
180 // Tail
181 if ( remainder > 0 ) {
182 ArrayRandomAccessIterator<byte> tailIt = input.At(input.Count - remainder);
183
184 ulong k = 0;
185
186 for ( int i = 0 ; i < remainder ; i++ ) {
187 byte b = Iterator.Advance<byte>(tailIt, i).Current;
188 k |= (ulong)b << (8 * i);
189 }
190
191 k *= 0x87c37b91114253d5UL;
192 k = RotateLeft64(k, 31);
193 k *= 0x4cf5ad432745937fUL;
194
195 h ^= (ulong)k;
196 }
197
198 h ^= (ulong)input.Count;
199 h = FMix64(h);
200
201 return new Hash64(h);
202 }
203

205 [MethodImpl(MethodImplOptions.AggressiveInlining)]
206 private static ulong RotateLeft64 ( ulong x, int r ) {
207 return (x << r) | (x >> (64 - r));
208 }
209
210 private static ulong FMix64 ( ulong k ) {
211 k ^= k >> 33;
212 k *= 0xff51afd7ed558ccd;
213 k ^= k >> 33;
214 k *= 0xc4ceb9fe1a85ec53;
215 k ^= k >> 33;
216
217 return k;
218 }
219
220 [MethodImpl(MethodImplOptions.AggressiveInlining)]
221 internal static uint RotateLeft ( uint x, byte r ) {
222 return x << (int)r | x >> 32 - (int)r;
223 }
224 internal static uint FMix ( uint h ) {
225 h = (uint)(((int)h ^ (int)(h >> 16)) * -2048144789);
226 h = (uint)(((int)h ^ (int)(h >> 13)) * -1028477387);
227 return h ^ h >> 16;
228 }
229 }

179 public bool Cost(T value, ref ulong cost, ulong budget) {
180 bool _ret = false;
181 // Priority Queue: (current cost, node)
182 var pq = new PriorityQueue<ICluster<T>, ulong>();
183 var visited = new Array<ICluster<T>>(128, 16);
184 var costs = new Map<ICluster<T>, ulong>();
185
186 // Start at this node with cost 0
187 pq.Enqueue(this, 0);
188 costs[this] = 0;
189
190 while ( pq.Count > 0 ) {
191 var current = pq.Dequeue();
192
193 // Skip if already visited
194 if ( visited.Contains(current) )
195 continue;
196
197 visited.Add(current);
198 ulong currentCost = costs[current];
199
200 // If current cost exceeds budget, abandon this path
201 if ( currentCost > budget )
202 continue;
203
204 // Found target?
205 if ( current.Value != null && current.Value.Equals(value) ) {
206 cost = currentCost;
207 _ret = true;
208 break;
209 }
210
211 // Traverse children
212 for ( int i = 0; i < current.Child.Size; i++ ) {
213 var childPair = current.Child[i]; // Pair<ICluster<T>, int>
214 ICluster<T> childNode = childPair.First;
215 uint edgeCost = childPair.Second;
216
217 if ( visited.Contains(childNode) )
218 continue;
219
220 ulong newCost = currentCost + (ulong)edgeCost;
221
222 // Only continue if within budget
223 if ( newCost > budget )
224 continue;
225
226 // Found a better path?
227 if ( !costs.ContainsKey(childNode) || newCost < costs[childNode] ) {
228 costs[childNode] = newCost;
229 pq.Enqueue(childNode, newCost);
230 }
231 }
232 }
233
234 // Return whether target was found within budget
235 return _ret;
236 }

241 public bool find(T pradicat, SearchType type, ulong budget, ref Map<ICluster<T>, uint> steps) {
242 // PriorityQueue: (cost, node)
243 var pq = new PriorityQueueEx<Triple<ICluster<T>, ulong, ulong>, ulong>();
244 var visited = new Array<ICluster<T>>(128, 16);
245
246
247 // Start: energie = budget, cost = 0
248 pq.Enqueue( new Triple<ICluster<T>, ulong, ulong>(this, budget, 0), 0);
249
250 while ( pq.Count > 0 ) {
251 var item = pq.Dequeue();
252 var current = item.First;
253 var energie = item.Second;
254 var cost = item.Third;
255
256 if ( visited.Contains(current) )
257 continue;
258
259 visited.Add(current);
260
261 // Schritt speichern
262 steps[current] = (uint)energie;
263
264 // Ziel gefunden?
265 if ( current.Value != null && (current.Value.Equals(pradicat) ) )
266 return true;
267
268 // Kinder durchgehen
269 for ( int i = 0; i < current.Child.Size; i++ ) {
270 var pair = current.Child[i];
271 var child = pair.First;
272 ulong gewicht = pair.Second;
273
274 if ( visited.Contains(child) )
275 continue;
276
277 ulong newEnergie = energie;
278 ulong newCost = cost;
279
280 switch ( type ) {
281 case SearchType.SubtractWeight:
282 if ( gewicht > newEnergie )
283 continue;
284 newEnergie -= (ulong)gewicht;
285 break;
286
287 case SearchType.SetToWeight:
288 newEnergie = (ulong)gewicht;
289 if ( newEnergie > budget )
290 continue;
291 break;
292
293 case SearchType.AccumulateCost:
294 newCost += (ulong)gewicht;
295 if ( newCost > budget )
296 continue;
297 break;
298 }
299
300 // Enqueue child with priority = accumulated cost
301 pq.Enqueue(new Triple<ICluster<T>, ulong, ulong>(child, newEnergie, newCost), newCost);
302 }
303 }
304 // Not found within budget/constraints
305 return false;
306 }

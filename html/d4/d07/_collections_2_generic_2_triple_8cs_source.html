30 [Serializable]
31#pragma warning disable CA1067
32 public struct Triple<TT, TU, TW> :
33 IEquatable<Triple<TT, TU, TW>>, ITuple
34#pragma warning restore CA1067
35 {
39 private TT m_first;
40
44 private TU m_second;
45
49 private TW m_third;
50
54 public TT First {
55 get => m_first;
56 set => m_first = value;
57 }
58
62 public TU Second {
63 get => m_second;
64 set => m_second = value;
65 }
66
70 public TW Third {
71 get => m_third;
72 set => m_third = value;
73 }
74
78 public readonly int Count => 3;
79
83 public Triple(TT first, TU second, TW third) {
84 m_first = first;
85 m_second = second;
86 m_third = third;
87 }
88
93 public readonly bool Equals(Triple<TT, TU, TW> other) {
94 return this.EqualFirst(other.First)
95 && this.EqualSecond(other.Second)
96 && this.EqualThird(other.Third);
97 }
98
104 public override string ToString() {
105 return string.Create(null, stackalloc char[384],
106 $"[{m_first}, {m_second}, {m_third}]");
107 }
108
112 public readonly bool EqualFirst(TT? other) {
113 if ( this.m_first == null )
114 throw new ArgumentNullException(nameof(other), "All Triple have a first");
115 return this.m_first.Equals(other);
116 }
117
121 public readonly bool EqualSecond(TU? other) {
122 if ( this.m_second == null )
123 throw new ArgumentNullException(nameof(other), "All Triple have a second");
124 return this.m_second.Equals(other);
125 }
126
130 public readonly bool EqualThird(TW? other) {
131 if ( this.m_third == null )
132 throw new ArgumentNullException(nameof(other), "All Triple have a third");
133 return this.m_third.Equals(other);
134 }
135
143 public object? Get(int index) {
144 if ( index < 0 || index >= Count )
145#pragma warning disable CA2201
146 throw new IndexOutOfRangeException("index");
147#pragma warning restore CA2201
148
149 return index switch
150 {
151 0 => m_first,
152 1 => m_second,
153 _ => m_third
154 };
155 }
156
161 public bool EqualFirst(object key) {
162 if ( key is TT typed )
163 return EqualFirst(typed);
164 return false;
165 }
166 }

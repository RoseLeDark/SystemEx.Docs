23namespace SystemEx.Collections.Generic {
32 public class SportedFixedArray<T> : FixedArray<T> {
36 private SortObjectFunc<T> m_sort;
37
42 private ICompared<T>? m_comparer;
43
48 public ICompared<T>? Comparer {
49 get => m_comparer;
50 set {
51 m_comparer = value;
52 if ( AutoSort ) Sort();
53 }
54 }
55
60 public SortObjectFunc<T> SortFunctions {
61 get => m_sort!;
62 set {
63 m_sort = value;
64 if ( AutoSort ) Sort();
65 }
66 }
67
70 public bool AutoSort { get; set; }

77 public SportedFixedArray(int size, SortObjectFunc<T> sort)
78 : base(size) {
79 m_sort = sort;
80 }

87 public SportedFixedArray(T[] e, SortObjectFunc<T> sort)
88 : base(e) {
89 m_sort = sort;

95 public override int Insert(int pos, T item) {
96 int _ret = base.Insert(pos, item);
97 if ( _ret >= 1 && AutoSort ) Sort();
98 return _ret;
99 }
100
104 public override int InsertRange(int pos, IEnumerable<T> items) {
105 int _ret = base.InsertRange(pos, items);
106 if ( _ret >= 1 && AutoSort ) Sort();
107 return _ret;
108 }
109
113 public override bool Remove() {
114 bool _ret = base.Remove();
115 if ( _ret && AutoSort ) Sort();
116 return _ret;
117 }
118
122 public void Sort() {
123 for ( int i = 0; i < Size - 1; i++ ) {
124 for ( int j = i + 1; j < Size; j++ ) {
125 CompareResult cmp = m_comparer != null
126 ? m_comparer.Compare(m_elements[i], m_elements[j])
127 : m_sort!(m_elements[i], m_elements[j]);
128
129 if ( cmp == CompareResult.AIsLargerB ) {
130 Swap(i, j);
131 }
132 }
133 }
134 }
135
140 [MethodImpl(MethodImplOptions.AggressiveInlining)]
141 private void Swap(int i, int j) {
142 var tmp = m_elements[i];
143 m_elements[i] = m_elements[j];
144 m_elements[j] = tmp;
145 }
146
151 public IArray<T> ToUnorderedArray() {
152 return new Array<T>(this.ToArray());
153 }
154 }

51 if ( boundary == 0 ) return (int)number;

480 public static uint SizeCalc(string value) {
481 string str = value.ToUpper(System.Globalization.CultureInfo.CurrentCulture).Trim();
482 uint multiplier = 1;
483
484 if ( str.EndsWith("G", StringComparison.OrdinalIgnoreCase) ) {
485 multiplier = 1024u * 1024u * 1024u;
486 str = str[..^1];
487 } else if ( str.EndsWith("M", StringComparison.OrdinalIgnoreCase) ) {
488 multiplier = 1024u * 1024u;
489 str = str[..^1];
490 } else if ( str.EndsWith("K", StringComparison.OrdinalIgnoreCase) ) {
491 multiplier = 1024u;
492 str = str[..^1];
493 } else if ( str.EndsWith("B", StringComparison.OrdinalIgnoreCase) ) {
494 multiplier = 1;
495 str = str[..^1];
496 }
497
498 return uint.TryParse(str, out uint size)
499 ? size * multiplier
500 : 512 * multiplier;
501 }

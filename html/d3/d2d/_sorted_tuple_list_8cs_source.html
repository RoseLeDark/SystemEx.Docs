134 public void Sort() {
135 for ( int i = 0; i < base.Count - 1; i++ ) {
136 for ( int j = i + 1; j < base.Count; j++ ) {
137
138 CompareResult cmp = m_comparer != null
139 ? m_comparer.Compare(m_elements[i], m_elements[j])
140 : m_sort!(m_elements[i], m_elements[j]);
141
142
143 if ( cmp == CompareResult.AIsLargerB ) {
144 Swap(i, j);
145 }
146 }
147 }
148 }
149
152 [MethodImpl(MethodImplOptions.AggressiveInlining)]
153 private void Swap(int i, int j) {
154 var tmp = m_elements[i];
155 m_elements[i] = m_elements[j];
156 m_elements[j] = tmp;
157 }

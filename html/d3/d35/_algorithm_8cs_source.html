247 public static bool LexicographicalCompare<T>(T[] a, T[] b, CompFunc<T> cmp) {
248 int n = System.Math.Min(a.Length, b.Length);
249
250 for ( int i = 0; i < n; i++ ) {
251 var r = cmp(a[i], b[i]);
252 if ( r == CompareResult.AIsSmallerB ) return true;
253 if ( r == CompareResult.AIsLargerB ) return false;
254 }
255
256 return a.Length < b.Length;

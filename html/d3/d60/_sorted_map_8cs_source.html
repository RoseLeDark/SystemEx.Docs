145 public void Sort() {
146 for ( int i = 0; i < Size - 1; i++ ) {
147 for ( int j = i + 1; j < Size; j++ ) {
148
149 CompareResult cmp = m_comparer != null
150 ? m_comparer.Compare(m_elements[i], m_elements[j])
151 : m_sort!(m_elements[i], m_elements[j]);
152
153 if ( cmp == CompareResult.AIsLargerB ) {
154 Swap(i, j);
155 }
156 }
157 }
158 }

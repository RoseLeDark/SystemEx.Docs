147 public void Sort() {
148 for ( int i = 0; i < Size - 1; i++ ) {
149 for ( int j = i + 1; j < Size; j++ ) {
150
151 CompareResult cmp = m_comparer != null
152 ? m_comparer.Compare(m_elements[i], m_elements[j])
153 : m_sort!(m_elements[i], m_elements[j]);
154
155 if ( cmp == CompareResult.AIsLargerB ) {
156 Swap(i, j);
157 }
158 }
159 }
160 }
161
165 [MethodImpl(MethodImplOptions.AggressiveInlining)]
166 private void Swap(int i, int j) {
167 var tmp = m_elements[i];
168 m_elements[i] = m_elements[j];
169 m_elements[j] = tmp;
170 }
171
176 public IMap<T, TU> ToUnorderedMap() {
177 Map<T, TU> map = [.. m_elements];
178 return map;
179 }
180
185 public override bool TryAdd(T k, TU v) {
186 var pair = base.TryAdd(k, v);
187 if ( AutoSort ) Sort();
188 return pair;
189 }
190 }

130 // MASK RANGE
131 // ---------------------------------------------------------------------
132
138 public static int MaskRange(int start, int length) {
139 if ( length <= 0 ) return 0;
140 if ( start < 0 || start > 31 ) return 0;
141 if ( length >= 32 ) return unchecked((int)0xFFFFFFFF);
142
143 int mask = (1 << length) - 1;
144 return mask << start;
145 }

90 public override ulong WriteRange(ulong start, ulong iend, byte[] data) {
91 if ( iend <= start ) return 0;
92 if ( iend > LongLength ) iend = LongLength;
93
94 ulong rangeLen = iend - start;
95 ulong writableLen = (ulong)data.Length < rangeLen ? (ulong)data.Length : rangeLen;
96
97 ulong totalWritten = 0;
98 ulong pos = start;
99 int dataOffset = 0;
100 ulong segmentSize = base.LongLength;
101
102 while ( totalWritten < writableLen ) {
103 ulong cacheIdx = pos / segmentSize;
104 ulong cacheOff = pos % segmentSize;
105 if ( cacheIdx >= (ulong)(m_caches.Length + 1) ) break;
106
107 ulong spaceInCache = segmentSize - cacheOff;
108 ulong remaining = writableLen - totalWritten;
109 uint chunkLen = (uint)((remaining < spaceInCache) ? remaining : spaceInCache);
110
111 // Fast path: ganzes data-Array passt und kein Offset nötig
112 if ( chunkLen == data.Length && dataOffset == 0 && cacheOff == 0 ) {
113
114 ulong written = (cacheIdx == 0)
115 ? base.WriteRange(cacheOff, cacheOff + (ulong)chunkLen, data)
116 : m_caches[cacheIdx - 1].WriteRange(cacheOff, cacheOff + (ulong)chunkLen, data);
117
118 if ( written == 0 ) break;
119
120 totalWritten += written;
121 pos += written;
122 dataOffset += (int)written;
123
124 } else {
125
126 m_segmentTemp.CopyFrom(data, (uint)dataOffset, 0, chunkLen);
127 // Hole internen Buffer (ToArray liefert m_elements)
128 byte[] internalBuf = m_segmentTemp.ToArray();
129
130 // Schreibe den Chunk in den jeweiligen Cache
131 ulong writtenChunk = (cacheIdx == 0)
132 ? base.WriteRange(cacheOff, cacheOff + (ulong)chunkLen, internalBuf)
133 : m_caches[cacheIdx - 1].WriteRange(cacheOff, cacheOff + (ulong)chunkLen, internalBuf);
134
135 if ( writtenChunk == 0 ) break;
136
137 totalWritten += writtenChunk;
138 pos += writtenChunk;
139 dataOffset += (int)writtenChunk;
140 }
141 }
142
143 return totalWritten;
144 }

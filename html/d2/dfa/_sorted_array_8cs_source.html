126 public void Sort() {
127 for ( int i = 0; i < Size - 1; i++ ) {
128 for ( int j = i + 1; j < Size; j++ ) {
129
130 CompareResult cmp = m_comparer != null
131 ? m_comparer.Compare(m_elements[i], m_elements[j])
132 : m_sort!(m_elements[i], m_elements[j]);
133
134 if ( cmp == CompareResult.AIsLargerB ) {
135 Swap(i, j);
136 }
137 }
138 }
139 }
140
146 [MethodImpl(MethodImplOptions.AggressiveInlining)]
147 private void Swap(int i, int j) {
148 var tmp = m_elements[i];
149 m_elements[i] = m_elements[j];
150 m_elements[j] = tmp;
151 }

157 public IArray<T> ToUnorderedArray() {

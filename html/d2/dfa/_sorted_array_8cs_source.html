124 public void Sort() {
125 for ( int i = 0; i < Size - 1; i++ ) {
126 for ( int j = i + 1; j < Size; j++ ) {
127
128 CompareResult cmp = m_comparer != null
129 ? m_comparer.Compare(m_elements[i], m_elements[j])
130 : m_sort!(m_elements[i], m_elements[j]);
131
132 if ( cmp == CompareResult.AIsLargerB ) {
133 Swap(i, j);
134 }
135 }
136 }

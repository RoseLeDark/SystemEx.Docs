165 public ColorHDR Lerp(ColorHDR value, float amount) {
166 // Quadratische Kurve
167 float q = amount * amount;

169 // Hue shortest-path interpolation
170 float dh = value.m_hue - m_hue;
171 dh = (dh + 540f) % 360f - 180f;
172
173 return new ColorHDR(
174 ClampHue(m_hue + dh * q),
175 m_saturation + (value.m_saturation - m_saturation) * q,
176 m_value + (value.m_value - m_value) * q
177 );

120 internal DeviceSharedBuffer( DeviceBuffer cache, TDeviceSharedBackend backend, int flags, object? config) {
121
122 m_cache = cache;
123
124 if ( cache.Type == Collections.Generic.CacheType.ToDevice )
125 m_type = SharedCacheType.WriteOnly;
126 else if ( cache.Type == Collections.Generic.CacheType.FromDevice )
127 m_type = SharedCacheType.ReadOnly;
128 else if ( cache.Type == Collections.Generic.CacheType.Both )
129 m_type = SharedCacheType.ReadWrite;
130 else
131 throw new NotSupportedException("DeviceBuffer is System Only ");
132
133 m_backend = backend;
134 m_hardwareBuffer = new object();
135 }
136
145 public bool Begin() {
146 m_cache.Lock();
147
148 if ( m_type != SharedCacheType.ReadOnly ) {
149 m_backend.CreateWriteHardwareBuffer(m_cache.ToArray(), out m_hardwareBuffer);
150 } else {
151 m_backend.CreateReadHardwareBuffer((int)m_cache.Length, out m_hardwareBuffer);
152 }
153
154 return true;
155 }
156
163 public void End() {
164
165 if ( m_type != SharedCacheType.ReadOnly ) {
166 byte[] cachetmp;
167 long startPos = m_backend.ReciveFromHardwareBuffer(out cachetmp, ref m_hardwareBuffer);
168
169 m_cache.WriteRange((ulong)startPos, cachetmp);
170 } else {
171 m_backend.CloseHardwareBuffer(ref m_hardwareBuffer);
172 }
173
174 m_cache.Unlock();
175 }
176 }

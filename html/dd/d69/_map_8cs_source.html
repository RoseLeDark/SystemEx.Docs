194 private bool Set(T key, TU value) {
195 for ( int i = 0; i < m_elements.Count; i++ ) {
196 if ( m_elements[i].EqualFirst(key) ) {
197
198 if ( m_elements[i].EqualSecond(value) )
199 return true;
200
201 int last = m_elements.Count - 1;
202 m_elements[i] = m_elements[last];
203 m_elements.RemoveAt(last);
204
205 m_elements.Add(new Pair<T, TU>(key, value));
206 return true;
207 }
208 }
209
210 return TryAdd(key, value);
211 }

260 public void Traverse(TraversMode mode, int startIndex, int endIndex, Action<Pair<T, TU>> func) {
261 int start = System.Math.Max(startIndex, 0);
262 int end = System.Math.Min(endIndex, m_elements.Count);
263
264 if ( mode == TraversMode.Forwards ) {
265 for ( int i = start; i < end; i++ )
266 func(m_elements[i]);
267 } else if ( mode == TraversMode.Backwards ) {
268 for ( int i = end; i >= start; i-- )
269 func(m_elements[i]);
270 }

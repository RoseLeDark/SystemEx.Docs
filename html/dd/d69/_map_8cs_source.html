192 private bool Set(T key, TU value) {
193 for ( int i = 0; i < m_elements.Count; i++ ) {
194 if ( m_elements[i].EqualFirst(key) ) {
195
196 if ( m_elements[i].EqualSecond(value) )

199 int last = m_elements.Count - 1;

201 m_elements.RemoveAt(last);

203 m_elements.Add(new Pair<T, TU>(key, value));
204 return true;
205 }

258 public void Traverse(TraversMode mode, int startIndex, int endIndex, Action<Pair<T, TU>> func) {

262 if ( mode == TraversMode.Forwards ) {
263 for ( int i = start; i < end; i++ )
264 func(m_elements[i]);

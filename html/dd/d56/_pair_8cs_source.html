18using SystemEx.Collections.Generic.Interfaces;
19
20namespace SystemEx.Collections.Generic {
30 [Serializable]
31 public struct Pair<T, TU> : IPair<T, TU> {
32
36 private T m_key;
37
41 private TU m_value;
42
46 public T First {
47 get => m_key;
48 set => m_key = value;
49 }
50
54 public TU Second {
55 get => m_value;
56 set => m_value = value;
57 }
58
62 public readonly int Count => 2;
63
69 public Pair(T first, TU second) {
70 m_key = first;
71 m_value = second;
72 }
73
80 public bool Equals(Pair<T, TU> other) {
81 return this.EqualFirst(other.First) && this.EqualSecond(other.Second);
82 }
83
88 public override string ToString() {
89 return string.Create(null, stackalloc char[256], $"[{m_key}, {m_value}]");
90 }
91
97 public readonly bool EqualFirst(T other) => this.m_key!.Equals(other);
98
104 public readonly bool EqualSecond(TU other) => this.m_value!.Equals(other);
105
115 public readonly object? Get(int index) {
116 if ( index < 0 || index >= Count )
117 throw new ArgumentOutOfRangeException(nameof(index));
118
119 return index == 0 ? m_key : m_value;
120 }
121
128 bool ITuple.EqualFirst(object key) {
129 if ( key is T typed )
130 return EqualFirst(typed);
131
132 return false;
133 }
134
135 }

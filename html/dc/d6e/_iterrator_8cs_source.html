188 public static IRandomAccessIterator<T> LowerBound<T>( IRandomAccessIterator<T> first, IRandomAccessIterator<T> last, T value, CompFunc<T> cmp) {
189 int count = Distance(first.Clone(), last.Clone());
190 IRandomAccessIterator<T> it = (IRandomAccessIterator<T>)first.Clone();
191
192 while ( count > 0 ) {
193 int step = count / 2;
194 IRandomAccessIterator<T> mid = (IRandomAccessIterator<T>)it.Clone();
195 mid.Advance(step);
196
197 if ( cmp(mid.Current, value) == CompareResult.AIsSmallerB ) {
198 it = mid;
199 it.Forward();
200 count -= step + 1;
201 } else {
202 count = step;
203 }
204 }
205
206 return it;
207 }
212 public static IRandomAccessIterator<T> UpperBound<T>( IRandomAccessIterator<T> first, IRandomAccessIterator<T> last, T value, CompFunc<T> cmp) {
213 int count = Distance(first.Clone(), last.Clone());
214 var it = first.Clone();
215
216 while ( count > 0 ) {
217 int step = count / 2;
218 IRandomAccessIterator<T> mid = (IRandomAccessIterator<T>)it.Clone();
219 mid.Advance(step);
220
221 if ( cmp(value, mid.Current) != CompareResult.AIsLargerB ) {
222 count = step;
223 } else {
224 it = mid;
225 it.Forward();
226 count -= step + 1;
227 }
228 }

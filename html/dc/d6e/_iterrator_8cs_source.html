170 public static int Distance<T>(IIterator<T> first, IIterator<T> last) {
171 int count = 0;
172 while ( !first.Equals(last) ) {
173 first.Forward();
174 count++;
175 }
176 return count;

296 public static IRandomAccessIterator<T> LowerBound<T>( IRandomAccessIterator<T> first, IRandomAccessIterator<T> last, T value, CompFunc<T> cmp) {
297 int count = Distance(first.Clone(), last.Clone());
298 IRandomAccessIterator<T> it = (IRandomAccessIterator<T>)first.Clone();
299
300 while ( count > 0 ) {
301 int step = count / 2;
302 IRandomAccessIterator<T> mid = (IRandomAccessIterator<T>)it.Clone();
303 mid.Advance(step);
304
305 if ( cmp(mid.Current, value) == CompareResult.AIsSmallerB ) {
306 it = mid;
307 it.Forward();
308 count -= step + 1;
309 } else {
310 count = step;
311 }
312 }
313
314 return it;
315 }

320 public static IRandomAccessIterator<T> UpperBound<T>( IRandomAccessIterator<T> first, IRandomAccessIterator<T> last, T value, CompFunc<T> cmp) {
321 int count = Distance(first.Clone(), last.Clone());
322 var it = first.Clone();
323
324 while ( count > 0 ) {
325 int step = count / 2;
326 IRandomAccessIterator<T> mid = (IRandomAccessIterator<T>)it.Clone();
327 mid.Advance(step);
328
329 if ( cmp(value, mid.Current) != CompareResult.AIsLargerB ) {
330 count = step;
331 } else {
332 it = mid;
333 it.Forward();
334 count -= step + 1;
335 }
336 }
337
338 return (IRandomAccessIterator <T> )it;
339 }

343 public static void Reverse<T>(IBidirectionalIterator<T> first, IBidirectionalIterator<T> last) {
344 last.Back();
345
346 while ( !first.Equals(last) && !first.IsEnd && !last.IsBegin ) {
347
348 T f = first.Current; T l = last.Current;
349 Algorithm.Swap(ref f, ref l);
350 first.Current = f; last.Current = l;
351
352 first.Forward();
353 last.Back();
354 }
355 }

167 public static int Distance<T>(IIterator<T> first, IIterator<T> last) {
168 int count = 0;
169 while ( !first.Equals(last) ) {
170 first.Forward();
171 count++;
172 }

191 public static IRandomAccessIterator<T> LowerBound<T>( IRandomAccessIterator<T> first, IRandomAccessIterator<T> last, T value, CompFunc<T> cmp) {
192 int count = Distance(first.Clone(), last.Clone());
193 IRandomAccessIterator<T> it = (IRandomAccessIterator<T>)first.Clone();
194
195 while ( count > 0 ) {
196 int step = count / 2;
197 IRandomAccessIterator<T> mid = (IRandomAccessIterator<T>)it.Clone();
198 mid.Advance(step);
199
200 if ( cmp(mid.Current, value) == CompareResult.AIsSmallerB ) {
201 it = mid;
202 it.Forward();
203 count -= step + 1;
204 } else {
205 count = step;
206 }
207 }
208
209 return it;
210 }

215 public static IRandomAccessIterator<T> UpperBound<T>( IRandomAccessIterator<T> first, IRandomAccessIterator<T> last, T value, CompFunc<T> cmp) {
216 int count = Distance(first.Clone(), last.Clone());
217 var it = first.Clone();
218
219 while ( count > 0 ) {
220 int step = count / 2;
221 IRandomAccessIterator<T> mid = (IRandomAccessIterator<T>)it.Clone();
222 mid.Advance(step);
223
224 if ( cmp(value, mid.Current) != CompareResult.AIsLargerB ) {
225 count = step;
226 } else {
227 it = mid;
228 it.Forward();
229 count -= step + 1;
230 }
231 }
232
233 return (IRandomAccessIterator <T> )it;
234 }

238 public static void Reverse<T>(IBidirectionalIterator<T> first, IBidirectionalIterator<T> last) {
239 last.Back();
240
241 while ( !first.Equals(last) && !first.IsEnd && !last.IsBegin ) {
242
243 T f = first.Current; T l = last.Current;
244 Algorithm.Swap(ref f, ref l);
245 first.Current = f; last.Current = l;
246
247 first.Forward();
248 last.Back();
249 }
250 }

35 public sealed class SimpleHasher : IHasher {
36
40 public Hash32 Compute ( Array<byte> input, Endian endian ) {
41 if ( input == null || input.Count == 0 )
42 return new Hash32(0);
43
44 int hash = 0;
45
46 // Simple deterministic byte loop
47 for ( int i = 0 ; i < input.Count ; i++ ) {
48 hash = (hash * 31) ^ input[i];
49 }
50
51 return new Hash32(hash);
52 }
53
57 public Hash64 ComputeLong ( Array<byte> input, Endian endian ) {
58 if ( input == null || input.Count == 0 )
59 return new Hash64(0);
60
61 long hash = 0;
62
63 // Larger multiplier for 64‑bit
64 for ( int i = 0 ; i < input.Count ; i++ ) {
65 hash = (hash * 1315423911L) ^ input[i];
66 }
67
68 return new Hash64(hash);
69 }

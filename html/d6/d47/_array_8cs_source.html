402 public virtual int Insert(int pos, T item) {

404 throw new ArgumentOutOfRangeException(nameof(pos));

418 public virtual int InsertRange(int pos, IEnumerable<T> items) {
419 if ( pos < 0 || pos > Size ) return 0;
420
421 // Materialisieren, damit wir Count kennen
422 var list = items as ICollection<T> ?? new List<T>(items);
423 int count = list.Count;
424
425 if ( count == 0 ) return 0;
426
427 // Prüfen ob genug Platz ist
428 int required = m_index + count;
429 if ( required > Size ) {
430 if ( !AutoGrow )
431 return 0;
432
433 // so lange wachsen, bis es passt
434 int newSize = Size;
435 while ( required > newSize )
436 newSize += GrowSize;
437
438 if ( !Resize(newSize) ) return -1;
439 }
440
441 // Platz schaffen: Block nach rechts schieben
442 for ( int i = m_index - 1; i >= pos; i-- )
443 m_elements[i + count] = m_elements[i];
444
445 // Elemente einfügen
446 int idx = pos;
447 int written = 0;
448 foreach ( var item in list ) {
449 m_elements[idx++] = item;
450 written++;
451 }
452 return written;
453 }

554 public int CopyFrom(T[] source, uint sourceOffset, uint destinationOffset, uint count) {
555 if ( source == null ) return 0;
556
557 int src = (int)sourceOffset;
558 int dst = (int)destinationOffset;
559
560 if ( dst > Size ) dst = Size;
561
562 int toCopy = System.Math.Min((int)count,
563 System.Math.Min(System.Math.Max(0, source.Length - src),
564 System.Math.Max(0, Size - dst)));
565
566 // Wenn nichts passt → prüfen ob wir wachsen müssen
567 if ( toCopy <= 0 ) {
568 // Prüfen ob AutoGrow aktiv ist
569 if ( !AutoGrow ) return 0;
570
571 // Wir brauchen mindestens count Bytes Platz ab dst
572 int required = dst + (int)count;
573
574 int newSize = Size;
575 while ( required > newSize )
576 newSize += GrowSize;
577
578 if ( !Resize(newSize) ) return 0;
579
580 // Nach Resize neu berechnen
581 toCopy = System.Math.Min((int)count, System.Math.Min(System.Math.Max(0, source.Length - src), System.Math.Max(0, Size - dst)));
582
583 if ( toCopy <= 0 )
584 return 0;
585 }
586
587 Buffer.BlockCopy(source, src, m_elements, dst, toCopy);
588 return toCopy;
589 }

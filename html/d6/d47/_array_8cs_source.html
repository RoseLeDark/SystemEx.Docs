349 public virtual int Insert(int pos, T item) {

365 public virtual int InsertRange(int pos, IEnumerable<T> items) {
366 if ( pos < 0 || pos > Size ) return 0;
367
368 // Materialisieren, damit wir Count kennen
369 var list = items as ICollection<T> ?? new List<T>(items);
370 int count = list.Count;
371
372 if ( count == 0 ) return 0;
373
374 // Prüfen ob genug Platz ist
375 int required = m_index + count;
376 if ( required > Size ) {
377 if ( !AutoGrow )
378 return 0;
379
380 // so lange wachsen, bis es passt
381 int newSize = Size;
382 while ( required > newSize )
383 newSize += GrowSize;
384
385 if ( !Resize(newSize) ) return -1;
386 }
387
388 // Platz schaffen: Block nach rechts schieben
389 for ( int i = m_index - 1; i >= pos; i-- )
390 m_elements[i + count] = m_elements[i];
391
392 // Elemente einfügen
393 int idx = pos;
394 int written = 0;
395 foreach ( var item in list ) {
396 m_elements[idx++] = item;
397 written++;
398 }
399 return written;
400 }

479 public int CopyTo(uint sourceOffset, T[] destination, uint destinationOffset, uint count) {

501 public int CopyFrom(T[] source, uint sourceOffset, uint destinationOffset, uint count) {
502 if ( source == null ) return 0;
503
504 int src = (int)sourceOffset;
505 int dst = (int)destinationOffset;
506
507 if ( dst > Size ) dst = Size;
508
509 int toCopy = System.Math.Min((int)count,
510 System.Math.Min(System.Math.Max(0, source.Length - src),
511 System.Math.Max(0, Size - dst)));
512
513 // Wenn nichts passt → prüfen ob wir wachsen müssen
514 if ( toCopy <= 0 ) {
515 // Prüfen ob AutoGrow aktiv ist
516 if ( !AutoGrow ) return 0;
517
518 // Wir brauchen mindestens count Bytes Platz ab dst
519 int required = dst + (int)count;
520
521 int newSize = Size;
522 while ( required > newSize )
523 newSize += GrowSize;
524
525 if ( !Resize(newSize) ) return 0;
526
527 // Nach Resize neu berechnen
528 toCopy = System.Math.Min((int)count, System.Math.Min(System.Math.Max(0, source.Length - src), System.Math.Max(0, Size - dst)));
529
530 if ( toCopy <= 0 )
531 return 0;
532 }
533
534 Buffer.BlockCopy(source, src, m_elements, dst, toCopy);
535 return toCopy;

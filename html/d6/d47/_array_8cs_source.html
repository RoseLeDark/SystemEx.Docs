369 public virtual int Insert(int pos, T item) {

371 throw new ArgumentOutOfRangeException(nameof(pos));
372
373 if ( m_index >= Size ) {
374 if ( !AutoGrow )

376 Resize(Size + GrowSize);

385 public virtual int InsertRange(int pos, IEnumerable<T> items) {
386 if ( pos < 0 || pos > Size ) return 0;
387
388 // Materialisieren, damit wir Count kennen
389 var list = items as ICollection<T> ?? new List<T>(items);
390 int count = list.Count;
391
392 if ( count == 0 ) return 0;
393
394 // Prüfen ob genug Platz ist
395 int required = m_index + count;
396 if ( required > Size ) {
397 if ( !AutoGrow )
398 return 0;
399
400 // so lange wachsen, bis es passt
401 int newSize = Size;
402 while ( required > newSize )
403 newSize += GrowSize;
404
405 if ( !Resize(newSize) ) return -1;
406 }
407
408 // Platz schaffen: Block nach rechts schieben
409 for ( int i = m_index - 1; i >= pos; i-- )
410 m_elements[i + count] = m_elements[i];
411
412 // Elemente einfügen
413 int idx = pos;
414 int written = 0;
415 foreach ( var item in list ) {
416 m_elements[idx++] = item;
417 written++;
418 }
419 return written;
420 }

521 public int CopyFrom(T[] source, uint sourceOffset, uint destinationOffset, uint count) {
522 if ( source == null ) return 0;
523
524 int src = (int)sourceOffset;
525 int dst = (int)destinationOffset;
526
527 if ( dst > Size ) dst = Size;
528
529 int toCopy = System.Math.Min((int)count,
530 System.Math.Min(System.Math.Max(0, source.Length - src),
531 System.Math.Max(0, Size - dst)));
532
533 // Wenn nichts passt → prüfen ob wir wachsen müssen
534 if ( toCopy <= 0 ) {
535 // Prüfen ob AutoGrow aktiv ist
536 if ( !AutoGrow ) return 0;
537
538 // Wir brauchen mindestens count Bytes Platz ab dst
539 int required = dst + (int)count;
540
541 int newSize = Size;
542 while ( required > newSize )
543 newSize += GrowSize;
544
545 if ( !Resize(newSize) ) return 0;
546
547 // Nach Resize neu berechnen
548 toCopy = System.Math.Min((int)count, System.Math.Min(System.Math.Max(0, source.Length - src), System.Math.Max(0, Size - dst)));

550 if ( toCopy <= 0 )
551 return 0;
552 }

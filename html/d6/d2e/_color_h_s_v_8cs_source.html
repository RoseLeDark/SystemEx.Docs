77 public virtual ColorHSV Addition(ColorHSV a) {
78 m_saturation = System.Math.Clamp(m_saturation + a.m_saturation, 0f, 1f);
79 m_value = System.Math.Clamp(m_value + a.m_value, 0f, 1f);
80 m_hue = ClampHue(m_hue + a.m_hue);
81 return this;

88 public virtual ColorHSV Subtraction(ColorHSV a) {
89 m_saturation = System.Math.Clamp(m_saturation - a.m_saturation, 0f, 1f);
90 m_value = System.Math.Clamp(m_value - a.m_value, 0f, 1f);
91 m_hue = ClampHue(m_hue - a.m_hue);
92 return this;
93 }
94
99 public virtual ColorHSV Multiplication(ColorHSV a) {
100 m_saturation = System.Math.Clamp(m_saturation * a.m_saturation, 0f, 1f);
101 m_value = System.Math.Clamp(m_value * a.m_value, 0f, 1f);
102 return this;
103 }
104
109 public virtual ColorHSV Division(ColorHSV a) {
110 if ( a.m_saturation != 0f )
111 m_saturation = System.Math.Clamp(m_saturation / a.m_saturation, 0f, 1f);
112
113 if ( a.m_value != 0f )
114 m_value = System.Math.Clamp(m_value / a.m_value, 0f, 1f);
115
116 return this;
117 }

125 public ColorHSV Lerp(ColorHSV value, float amount) {
126 float q = amount * amount;
127
128 // Shortest-path hue interpolation
129 float dh = value.m_hue - m_hue;
130 dh = (dh + 540f) % 360f - 180f;
131
132 return new ColorHSV(
133 ClampHue(m_hue + dh * q),
134 m_saturation + (value.m_saturation - m_saturation) * q,
135 m_value + (value.m_value - m_value) * q
136 );
137 }

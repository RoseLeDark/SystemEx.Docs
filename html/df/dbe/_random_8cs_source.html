85 public static string RandPassword(int length, FixedArray<char> allowed, Endian endian) {
86
87 StringBuilder sb = new StringBuilder(length);
88
89 int i = 0, d = 0;
90 char c = '\0';
91 while(i < length ) {
92 c = RandChar((char)0, (char)short.MaxValue, endian);
93 if ( allowed.TryGet(c, out d) ) {
94 i++;
95 sb.Append(c);
96 }
97 }
98
99 return sb.ToString();
100 }

159 public static byte RandByte(byte min, byte max, Endian endian) {
160 if ( min > max ) {
161 byte tmp = min;
162 min = max;
163 max = tmp;
164 }
165
166 byte _h = (byte)GetArray(1)[0];
167 int range = (max - min) + 1;
168
169 return (byte)(min + (_h % range));
170 }

356 public static ulong RandULong(ulong min, ulong max, Endian endian) {
357 if ( min > max )
358 throw new ArgumentException("min must be <= max");
359
360 // 1. 8 zufällige Bytes erzeugen
361 byte[] _array_h = GetArray(4);
362 byte[] _array_l = GetArray(4);
363
364 // 3. Range-Mapping (raw % range)
365 ulong range = (ulong)(max - min + 1);
366 ulong value = ((_array_h.ToUInt(endian) << 32) | _array_l.ToUInt(endian)) % range;
367
368 // 4. Offset hinzufügen
369 return (ulong)(min + (ulong)value);
370 }

375 public static long RandLong(long min, long max, Endian endian) {
376 if ( min > max ) {
377 long tmp = min;
378 min = max;
379 max = tmp;
380 }
381
382 int _h = GetArray(4).ToInt(endian);
383 uint _l = GetArray(4).ToUInt(endian);
384
385 ulong raw = ((ulong)(uint)_h << 32) | _l;
386
387 ulong range = (ulong)(max - min) + 1;
388
389 long value = (long)(raw % range);
390
391 return min + value;
392 }

407 public static byte[] GetArray(int size, byte min, byte max) {
408 if ( min > max )
409 throw new ArgumentException("min must be <= max");
410
411 byte[] buffer = new byte[size];
412
413 for ( int i = 0; i < size; i++ ) {
414 buffer[i] = (byte)r.Next(min, max + 1);
415 }
416
417 return buffer;
418 }

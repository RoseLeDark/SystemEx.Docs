260 public void Traverse(TraversMode mode, int startIndex, int endIndex, Action<Pair<T, TU>> func) {

262 int end = System.Math.Min(endIndex, m_count);

264 if ( mode == TraversMode.Forwards ) {
265 for ( int i = start; i < end; i++ )
266 func(m_elements[i]);

73 public override int InsertRange(int pos, IEnumerable<T> items) {
74 if ( pos < 0 ) return 0;
75
76 // Materialisieren, damit wir Count kennen
77 var list = items as ICollection<T> ?? new List<T>(items);
78 int count = list.Count;
79
80 if ( count == 0 ) return 0;
81
82 // Prüfen ob genug Platz ist
83 int space = Size - pos; // wie viel passt ab pos?
84 int toWrite = count > space ? space : count;
85
86 int idx = pos;
87 int written = 0;
88 foreach ( var item in list ) {
89 if ( written >= toWrite )
90 break;
91
92 m_elements[idx++] = item;
93 written++;
94 }
95
96 return written;
97 }

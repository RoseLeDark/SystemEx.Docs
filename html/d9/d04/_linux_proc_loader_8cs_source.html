103 public static string GetLibaryPath ( string somodule ) {
104 string _ret = NO_PATH;
105
106 // Check /usr/lib
107 _ret = GetFromPath("/usr/lib/", somodule);
108
109 // Check LD_LIBRARY_PATH
110 if ( _ret == NO_PATH ) {
111 string? env = Environment.GetEnvironmentVariable("LD_LIBRARY_PATH");
112 if ( env != null ) {
113 var paths = env.Split(':');
114 foreach ( string path in paths ) {
115 string candidate = GetFromPath(path, somodule);
116 if ( candidate != NO_PATH ) {
117 _ret = candidate;
118 break;
119 }
120 }
121 }
122 }
123
124 return _ret;
125 }

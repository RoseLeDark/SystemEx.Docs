18using SystemEx.Collections.Generic.Interfaces;
19
20namespace SystemEx.Collections.Generic {
40 public class TypeBuffer<T> : ITypeBuffer<T>
41 where T : unmanaged {
42 private readonly Cache m_cache;
43 private readonly int m_stride;
44 private readonly int m_length;
45 private Endian m_endian;
46
50 public int Length => m_length;
51
55 public Endian Endian => m_endian;
56
67 public T this[int i] {
68 get => Read(i * m_stride, m_stride, m_endian);
69 set => Write(value, i * m_stride, m_stride, m_endian);
70 }
71
80 public TypeBuffer ( int length, Endian endian ) {
81 m_length = length;
82 unsafe { m_stride = sizeof(T); }
83 m_cache = new Cache(length * m_stride, CacheType.OnlySystem);
84 m_endian = endian;
85 }
86
95 public TypeBuffer ( TypeBuffer<T> other ) {
96 m_length = other.m_length;
97 m_stride = other.m_stride;
98 m_cache = new Cache(other.m_cache);
99 m_endian = other.m_endian;
100 }
101
109 public void Clear () {
110 m_cache.SetZero();
111 }
112
117 public ITypeBuffer<T> Clone () {
118 return new TypeBuffer<T>(this);
119 }
120
124 public void Zero () {
125 m_cache.SetZero();
126 }
127
138 public ITypeBuffer<T> Fill ( T value ) {
139 for ( int i = 0 ; i < m_length ; i++ )
140 Write(value, i * m_stride, m_stride, m_endian);
141 return this;
142 }
143
157 public int Write ( T value, int offset, int count, Endian endian ) {
158 var data = Conversion.ToBytes<T>(value, endian);
159 return m_cache.Write(data, offset, count);
160 }
161
174 public T Read ( int offset, int count, Endian endian ) {
175 byte[] buffer = new byte[m_stride];
176 int read = m_cache.Read(buffer, offset, count);
177 return Conversion.FromBytes<T>(buffer, endian);
178 }
179 }

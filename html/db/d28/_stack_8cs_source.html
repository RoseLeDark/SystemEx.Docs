220 public int PushRange(T[] items, uint LayerID = 100) {
221 if ( LayerID >= 4 ) return -1;
222 if ( items == null ) return -1;
223
224 bool layered = LayerID <= m_stackLayer.Length;
225 int i = 0;
226
227 for ( i = 0; i < items.Length; i++ ) {
228 if ( layered ) {
229 if ( !Push(items[i], LayerID) ) break;

231 if ( !Push(items[i]) ) break;

244 public T[]? PopRange(int size, uint LayerID = 100) {

249

251
252 for ( int i = 0; i < size; i++ ) {
253 if ( layered ) {
254 if ( !Pop(out item, LayerID) ) break;
255 } else {
256 if ( !Pop(out item) ) break;

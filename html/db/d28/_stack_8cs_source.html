153 public Stack(int size) {
154 m_elements = new T[size];
155 m_end = 0;
156 m_start = m_elements.Length - 1;
157 m_current = size-1;
158
159 for (int i = 0; i < m_stackLayer.Length; i++) {
160 m_stackLayer[i] = new StackLayer(m_end, m_start, m_current);
161 m_stackLayer[i].Enable = false;
162 m_stackLayer[i].EndMarker = m_end;
163 m_stackLayer[i].StartMarker = m_start;
164 m_stackLayer[i].Current = m_current;
165 }

172 public bool Push(T element) {

182 public bool Push(T item, uint LayerID) {
183 if ( LayerID >= 4 ) return false;
184 if ( !m_stackLayer[LayerID].Enable ) return false;
185
186 if ( m_stackLayer[LayerID].Current == m_stackLayer[LayerID].EndMarker )
187 return false;
188
189 m_elements[m_stackLayer[LayerID].Current] = item;
190 m_stackLayer[LayerID].Current--;
191 return true;
192 }

199 public bool Pop(out T? item) {

201 item = m_elements[m_current];
202 m_current++;
203 return true;
204 }

221 public int PushRange(T[] items, uint LayerID = 100) {
222 if ( LayerID >= 4 ) return -1;
223 if ( items == null ) return -1;
224
225 bool layered = LayerID <= m_stackLayer.Length;
226 int i = 0;
227
228 for ( i = 0; i < items.Length; i++ ) {
229 if ( layered ) {
230 if ( !Push(items[i], LayerID) ) break;
231 } else {
232 if ( !Push(items[i]) ) break;
233 }
234 }
235
236 return i;
237 }

245 public T[]? PopRange(int size, uint LayerID = 100) {
246 if ( LayerID >= 4 ) return null;
247
248 List<T> popped = new List<T>();
249 T? item;
250
251 bool layered = LayerID <= m_stackLayer.Length;
252
253 for ( int i = 0; i < size; i++ ) {
254 if ( layered ) {
255 if ( !Pop(out item, LayerID) ) break;
256 } else {
257 if ( !Pop(out item) ) break;
258 }
259
260 if ( item != null ) popped.Add(item);
261 }
262
263 return popped.ToArray();
264 }

268 public bool Pop(out T? item, uint LayerID) {
269 if ( LayerID >= 4 ) { item = default; return false; }
270 if ( (m_stackLayer[LayerID].Enable != true) ) { item = default; return false; }
271 // IsEmpty
272 if ( m_stackLayer[LayerID].Current == m_stackLayer[LayerID].StartMarker - 1 ) { item = default; return false; }
273 item = m_elements[m_stackLayer[LayerID].Current];
274
275 m_stackLayer[LayerID].Current++;
276
277 return true;
278
279 }

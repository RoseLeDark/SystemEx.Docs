421 public void Swap(ref Node<T> other) {
422 if ( this == other )
423 return;
424
425 // Backup A
426 var aPrev = this.Prev;
427 var aNext = this.Next;
428 var aValue = this.Value;
429
430 // Backup B

432 var bNext = other.Next;
433 var bValue = other.Value;
434
435 // Copy B into A
436 this.Prev = bPrev;
437 this.Next = bNext;
438 this.Value = bValue;
439
440 // Copy A into B
441 other.Prev = aPrev;
442 other.Next = aNext;
443 other.Value = aValue;
444
445 // Fix neighbors of A
446 this.Prev.Next = this;
447 this.Next.Prev = this;
448
449 // Fix neighbors of B
450 other.Prev.Next = other;
451 other.Next.Prev = other;

537 private static void TraversPreorder(Node<T> node, Action<Node<T>> action) {
538 if ( node == null ) return;

540 action(node);
541
542 // Childs

544 TraversPreorder(node.m_pChilds[i], action);
545

555

345 public Node<T> GetAt(int index, out int r) {
346 Node<T> temp = this;
347 int n = index;
348
349 if ( n > 0 ) {
350 while ( n != 0 && temp.HasNext ) {
351 temp = temp.Next;
352 n--;
353 }
354 } else if ( n < 0 ) {
355 while ( n != 0 && temp.HasPrev ) {
356 temp = temp.Prev;
357 n++;
358 }
359 }
360
361 r = n;
362 return temp;

367 public Node<T> Insert(Node<T> pNext) {
368 // detach this if already linked
369 if ( HasNext || HasPrev )
370 remove();
371
372 // now insert
373 m_pChilds[NEXT] = pNext;
374 m_pChilds[PREV] = pNext.Prev;
375 pNext.Prev.m_pChilds[NEXT] = this;
376 pNext.m_pChilds[PREV] = this;

378 return this;

432 public void Swap(ref Node<T> other) {
433 if ( this == other )
434 return;
435
436 // Backup A
437 var aPrev = this.Prev;
438 var aNext = this.Next;
439 var aValue = this.Value;
440
441 // Backup B
442 var bPrev = other.Prev;
443 var bNext = other.Next;
444 var bValue = other.Value;
445
446 // Copy B into A
447 this.Prev = bPrev;
448 this.Next = bNext;
449 this.Value = bValue;
450
451 // Copy A into B
452 other.Prev = aPrev;
453 other.Next = aNext;
454 other.Value = aValue;
455
456 // Fix neighbors of A
457 this.Prev.Next = this;
458 this.Next.Prev = this;
459
460 // Fix neighbors of B
461 other.Prev.Next = other;
462 other.Next.Prev = other;
463 }

479 public virtual void Travers(TraversOrder order, Action<Node<T>> action) {
480 switch ( order ) {
481 case TraversOrder.ListOrder:
482 TraversListForward(action);
483 break;
484
485 case TraversOrder.ReservListOrder:
486 TraversListBackward(action);
487 break;
488
489 case TraversOrder.Preorder:
490 TraversPreorder(this, action);
491 break;
492
493 case TraversOrder.Inorder:
494 break;
495
496 case TraversOrder.Postorder:
497 TraversPostorder(this, action);
498 break;
499 }

531 private void TraversListForward(Action<Node<T>> action) {
532 var temp = this;
533 while ( temp.HasNext ) {
534 action(temp);
535 temp = temp.Next;
536 }
537 action(temp); // letztes Element
538 }

540 private void TraversListBackward(Action<Node<T>> action) {
541 var temp = this;
542 while ( temp.HasPrev ) {
543 action(temp);
544 temp = temp.Prev;
545 }
546 action(temp); // erstes Element
547 }
548 private static void TraversPreorder(Node<T> node, Action<Node<T>> action) {
549 if ( node == null ) return;
550
551 action(node);
552
553 // Childs
554 for ( int i = 2; i < node.m_pChilds.Size; i++ )
555 TraversPreorder(node.m_pChilds[i], action);
556
557 // Siblings
558 for ( int i = 0; i < node.m_pSiblings.Size; i++ )
559 TraversPreorder(node.m_pSiblings[i], action);
560 }
561 private static void TraversPostorder(Node<T> node, Action<Node<T>> action) {
562 if ( node == null ) return;
563
564 for ( int i = 2; i < node.m_pChilds.Size; i++ )
565 TraversPostorder(node.m_pChilds[i], action);
566
567 for ( int i = 0; i < node.m_pSiblings.Size; i++ )
568 TraversPostorder(node.m_pSiblings[i], action);
569
570 action(node);
571 }

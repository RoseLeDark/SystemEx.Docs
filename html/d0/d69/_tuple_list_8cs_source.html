22namespace SystemEx.Collections.Generic {
30 public class TupleList : IEnumerable<ITuple>, IEnumerable {
34 internal List<ITuple> m_elements;
35
39 public int Count => m_elements.Count;
40
44 public ITuple this[int index] {
45 get => m_elements[index];
46 set => m_elements[index] = value;
47 }
48
52 public TupleList() {
53 m_elements = new List<ITuple>();
54 }
55
59 public TupleList(int size) {
60 m_elements = new List<ITuple>(size);
61 }
62
65 public TupleList(IEnumerable<ITuple> collection) {
66 m_elements = new List<ITuple>(collection);
67 }
68
74 public List<TU> GetAll<TU>() where TU : ITuple {
75 List<TU> _ret = new List<TU>();
76
77 foreach ( var item in m_elements ) {
78 if ( item == null ) continue;
79 if ( item is TU ) _ret.Add((TU)item);
80 }
81
82 return _ret;
83 }
84
89 public List<ITuple> GetByCount(byte count) {
90 List<ITuple> _ret = new List<ITuple>();
91
92 foreach ( var item in m_elements ) {
93 if ( item == null ) continue;
94 if ( item.Count == count ) _ret.Add(item);
95 }
96
97 return _ret;
98 }
99
103 public virtual void Add(ITuple tuple) {
104 m_elements.Add(tuple);
105 }
106
110 public virtual void AddRange(IEnumerable<ITuple> items) {
111 m_elements.AddRange(items);
112 }
113
117 public void Clear() => m_elements.Clear();
118
122 public bool Contains(ITuple item) => m_elements.Contains(item);
123
127 public void CopyTo(ITuple[] array, int arrayIndex) =>
128 m_elements.CopyTo(array, arrayIndex);
129
133 public virtual bool Remove(ITuple item) => m_elements.Remove(item);
134
138 public IEnumerator<ITuple> GetEnumerator() => m_elements.GetEnumerator();
139
140 IEnumerator IEnumerable.GetEnumerator() => GetEnumerator();
141
145 public int IndexOf(ITuple item) => m_elements.IndexOf(item);
146
150 public virtual void Insert(int index, ITuple item) =>
151 m_elements.Insert(index, item);
152
156 public virtual void RemoveAt(int index) =>
157 m_elements.RemoveAt(index);
158 }

18using System.Collections;
19using System.Numerics;
20using SystemEx.Collections.Generic.Interfaces;
21
22
23namespace SystemEx {
30 public class NumberRange<T> : IRange<T>, IEquatable<NumberRange<T>>, IEnumerable<T>
31 where T : INumber<T> {
33 public T From { get; set; }
35 public T To { get; set; }
36
40 public NumberRange () {
41 this.From = T.Zero;
42 this.To = T.One;
43 }
44
49 public NumberRange (T start, T end) {
50 this.From = start;
51 this.To = end;
52 }
53
56 public T Length => this.To - From;
60 public bool IsValid => this.From <= this.To;
64 public bool IsSame => this.From == this.To;
68 public IForwardIterator<T> Begin => new NumberRangeIterator<T>(this.From, this.To);
72 public IForwardIterator<T> End => new NumberRangeIterator<T>(this.To + T.One, this.To);
73
75 public override int GetHashCode() =>
76 HashCode.Combine(this.From, this.To);
77
79 public override bool Equals(object? obj) =>
80 Equals(obj as NumberRange<T>);
81
87 public bool Equals(NumberRange<T>? other) =>
88 other is not null &&
89 From.Equals(other.From) &&
90 this.To.Equals(other.To);
91
98 public bool Contains(T x) => (x - this.To) * (x - this.From) <= T.Zero;
99
104 public NumberRange<T> Normalize() =>
105 (this.From <= this.To) ? this : new NumberRange<T>(this.To, this.From);
106
108 IEnumerator IEnumerable.GetEnumerator() =>
109 GetEnumerator();
110
117 public IRange<T> GetRange(T? tstart, T? tend) {
118 return new NumberRange<T>((tstart != null) ? tstart : this.From,
119 (tend != null) ? tend : this.To);
120 }
121
126 public IEnumerator<T> GetEnumerator() {
127 for ( T i = From; i <= this.To; i += T.One )
128 yield return i;
129 }
130
136 public bool Overlaps(IRange<T> other) {
137 bool result = false;
138
139 if ( IsValid && other.IsValid )
140 result = !(other.To < From || other.From > this.To);
141
142 return result;
143 }
144
152 public IRange<T>? Intersect(IRange<T> other) {
153 NumberRange<T>? _ret = null;
154
155 if ( Overlaps(other) ) {
156 _ret = new NumberRange<T>(T.Max(this.From, other.From),
157 T.Min(this.To, other.To));
158 }
159
160 return _ret;
161 }
162
168 public bool IsAdjacent(IRange<T> other) {
169 bool result = false;
170
171 if ( IsValid && other.IsValid )
172 result =
173 this.To == other.From ||
174 other.To == this.From;
175
176 return result;
177 }
178
185 public IRange<T>? Union( IRange<T> other) {
186 IRange<T>? result = null;
187
188 if ( Overlaps(other) || IsAdjacent(other) ) {
189 T start = T.Min(this.From, other.From);
190 T end = T.Max(this.To, other.To);
191 result = new NumberRange<T>(start, end);
192 }
193
194 return result;
195 }
196
200 public override string ToString() {
201 return (IsValid) ?
202 string.Create(null, stackalloc char[256], $"Range: [{this.From} ... $[{this.To}]") :
203 string.Create(null, stackalloc char[256], $"Not Valid Range: [{this.From} ... $[{this.To}]");
204 }
205 }
206}

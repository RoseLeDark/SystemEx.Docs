106 public class PriorityQueueEx<T, TU>
107 where TU : INumber<TU> {
111 private readonly SortedMap<T, TU> m_map;
112
116 public int Count => m_map.Size;
117
121 public Pair<T, TU> Current => m_map[0];
122
126 public PriorityQueueExIterator<T, TU> Iterator =>
127 new PriorityQueueExIterator<T, TU>(this.Clone());
128
132 public ICompared<IPair<T, TU>>? Comparer {
133 get => m_map.Comparer;
134 set => m_map.Comparer = value;
135 }
136
140 public SortFunc<T, TU>? SortFunctions {
141 get => m_map.SortFunctions;
142 set => m_map.SortFunctions = value;
143 }
144
145
149 public Pair<T, TU>? Min => m_map[0];
150
154 public Pair<T, TU>? Max => m_map[m_map.Size - 1];
155
159 public Pair<T, TU>? Median => GetClosestToMedian();
160
164 public PriorityQueueEx() {
165 m_map = new SortedMap<T, TU>(SortByPriority);
166 }
167
171 public PriorityQueueEx(ICompared<IPair<T, TU>> comparer) {
172 m_map = new SortedMap<T, TU>(comparer);
173 }
174
178 public PriorityQueueEx(SortFunc<T, TU> sort) {
179 m_map = new SortedMap<T, TU>(sort);
180 }
181
185 public PriorityQueueEx(IEnumerable<Pair<T, TU>> map) {
186 m_map = new SortedMap<T, TU>(map, SortByPriority);
187 }
188
192 public PriorityQueueEx<T, TU> Clone() {
193 var c = new PriorityQueueEx<T, TU>(m_map);
194 c.SortFunctions = SortFunctions;

200
204 public bool Enqueue(T item, TU priority) {
205 return m_map.TryAdd(item, priority);
206 }
207
211 public T Peek() {
212 return m_map[0].First;

218 public T Dequeue() {
219 var p = m_map[0];

234 private Pair<T, TU>? GetClosestToMedian() {
235 int n = m_map.Size;
236 if ( n == 0 )
237 return null;
238
239 SortedArray<TU> prios = new SortedArray<TU>(n, SortByPriorityArray);
240 for ( int i = 0; i < n; i++ )
241 prios.Add(m_map[i].Second);
242
243 TU median;
244 if ( (n & 1) == 1 ) {
245 median = prios[n / 2];
246 } else {
247 TU a = prios[(n / 2) - 1];
248 TU b = prios[n / 2];
249 median = (a + b) / TU.CreateChecked(2);
250 }
251
252 Pair<T, TU> best = m_map[0];
253 TU bestDist = Abs(best.Second - median);
254
255 for ( int i = 1; i < n; i++ ) {
256 var p = m_map[i];
257 TU dist = Abs(p.Second - median);
258
259 if ( dist < bestDist ) {
260 best = p;
261 bestDist = dist;
262 }
263 }
264
265 return best;
266 }
267
268 private static TU Abs(TU value) {
269 return value < TU.Zero ? -value : value;
270 }
271
272 private static CompareResult SortByPriorityArray(TU a, TU b) {
273 if ( a < b ) return CompareResult.AIsSmallerB;
274 if ( a > b ) return CompareResult.AIsLargerB;
275 return CompareResult.Equal;
276 }
277
278 private static CompareResult SortByPriority(Pair<T, TU> a, Pair<T, TU> b) {
279 if ( a.Second < b.Second ) return CompareResult.AIsSmallerB;
280 if ( a.Second > b.Second ) return CompareResult.AIsLargerB;
281 return CompareResult.Equal;
282 }
283 }

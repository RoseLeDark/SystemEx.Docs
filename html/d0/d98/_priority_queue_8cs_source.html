222 private Pair<T, TU>? GetClosestToMedian() {
223 int n = m_map.Size;
224 if ( n == 0 )
225 return null;
226
227 SortedArray<TU> prios = new SortedArray<TU>(n, SortByPriorityArray);
228 for ( int i = 0; i < n; i++ )
229 prios.Add(m_map[i].Second);
230
231 TU median;
232 if ( (n & 1) == 1 ) {
233 median = prios[n / 2];
234 } else {
235 TU a = prios[(n / 2) - 1];
236 TU b = prios[n / 2];
237 median = (a + b) / TU.CreateChecked(2);
238 }
239
240 Pair<T, TU> best = m_map[0];
241 TU bestDist = Abs(best.Second - median);
242
243 for ( int i = 1; i < n; i++ ) {
244 var p = m_map[i];
245 TU dist = Abs(p.Second - median);
246
247 if ( dist < bestDist ) {
248 best = p;
249 bestDist = dist;
250 }
251 }

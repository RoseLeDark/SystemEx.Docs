224 private Pair<T, TU>? GetClosestToMedian() {
225 int n = m_map.Size;
226 if ( n == 0 )
227 return null;
228
229 SortedArray<TU> prios = new SortedArray<TU>(n, SortByPriorityArray);
230 for ( int i = 0; i < n; i++ )
231 prios.Add(m_map[i].Second);
232
233 TU median;
234 if ( (n & 1) == 1 ) {
235 median = prios[n / 2];
236 } else {
237 TU a = prios[(n / 2) - 1];
238 TU b = prios[n / 2];
239 median = (a + b) / TU.CreateChecked(2);
240 }
241
242 Pair<T, TU> best = m_map[0];
243 TU bestDist = Abs(best.Second - median);
244
245 for ( int i = 1; i < n; i++ ) {
246 var p = m_map[i];
247 TU dist = Abs(p.Second - median);
248
249 if ( dist < bestDist ) {
250 best = p;
251 bestDist = dist;
252 }
253 }
254
255 return best;
256 }

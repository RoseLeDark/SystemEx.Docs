208 public virtual Task WriteAsync(ICache src, CancellationToken cancellationToken = default) {
209 if ( !m_cache.CanWrite ) throw new InvalidOperationException("dest is Locked");
210 if ( !src.CanRead ) throw new InvalidOperationException("src is Locked");
211
212 if ( cancellationToken.IsCancellationRequested )
213 return Task.FromCanceled(cancellationToken);
214
215 try {
216 const int CHUNK = 64 * 1024; // 64 KiB
217 ulong total = (ulong)src.Length;
218 ulong pos = 0;
219
220 while ( pos < total ) {
221 if ( cancellationToken.IsCancellationRequested )
222 return Task.FromCanceled(cancellationToken);
223
224 int toRead = (int)System.Math.Min((ulong)CHUNK, total - pos);
225 byte[]? chunk = src.ReadRange(pos, (uint)toRead);
226
227 if ( chunk == null || chunk.Length == 0 )
228 break;
229
230 m_cache.WriteRange(pos, chunk);
231 pos += (ulong)chunk.Length;
232 }
233
234 return Task.CompletedTask;
235 } catch ( OperationCanceledException oce ) {
236 return Task.FromCanceled(oce.CancellationToken);
237 } catch ( Exception ex ) {
238 return Task.FromException(ex);
239 }
240 }
